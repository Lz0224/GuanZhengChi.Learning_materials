
#npm 使用

```
npm install xxxx  //本地安装
npm install xxx@version //本地安装指定的版本

npm install  -g  xxx  //全局安装
npm uninstall  -g xxx  //卸载全局安装

npm update  xxx  //更新


npm install xxx@latest  //本地安装最新版本
npm ls // 查看npm本地目录
npm ls -g //查看npm全局目录

npm --help 帮助文档


npm init --yes //产生一个package.json文件

npm install jquery  --save  //  --save 将jquery作为项目依赖填写到package.json
npm install jquery -S //  -S将jquery作为项目依赖填写到package.json
npm install xxx  --save-dev  //将xxx作为开发工具的模块依赖

npm install  //直接安装package.json中的依赖模块

```
